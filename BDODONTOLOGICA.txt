  CREATE TABLE CLINICA(
  ID_CLINICA VARCHAR(15) PRIMARY KEY NOT NULL,
  NOMBRE_CLINICA VARCHAR(25) NOT NULL
  )

  CREATE TABLE LOCALES(
  ID_LOCALES VARCHAR(15) PRIMARY KEY NOT NULL,
  NOMBRE_LOCAL VARCHAR(25) NOT NULL,
  CIUDAD_LOCAL VARCHAR(25) NOT NULL,
  CALLE_LOCAL VARCHAR(10) NOT NULL,
  NUMERO_LOCAL VARCHAR(10) NOT NULL,
  ID_CLINICA_FK VARCHAR(15)
  FOREIGN KEY (ID_CLINICA_FK) REFERENCES CLINICA (ID_CLINICA)
  )

  CREATE TABLE CONSULTORIOS(
  ID_NRO_CONSULTORIO VARCHAR(15) PRIMARY KEY NOT NULL,
  ID_LOCAL_FK VARCHAR(15) NOT NULL
  FOREIGN KEY (ID_LOCAL_FK) REFERENCES LOCALES (ID_LOCALES)
  )

  CREATE TABLE EQUIPAMIENTOS (
  ID_EQUIPAMIENTO VARCHAR(15) PRIMARY KEY NOT NULL,
  TIPO_EQUIPAMIENTO VARCHAR(25) NOT NULL,
  NRO_SERIE VARCHAR(15) NOT NULL,
  MANTENIMIENTO DATE NOT NULL,
  ID_NRO_CONSULTORIO_FK VARCHAR(15)
  FOREIGN KEY (ID_NRO_CONSULTORIO_FK) REFERENCES CONSULTORIOS (ID_NRO_CONSULTORIO)

   )

   CREATE TABLE AFILIADOS(
   ID_AFILIADO VARCHAR(15) PRIMARY KEY NOT NULL,
   NOMBRE_AFILIADO VARCHAR(25) NOT NULL,
   APELLIDO_AFILIADO VARCHAR(25) NOT NULL,
   TIPO_DOCUMENTO_AFILIADO VARCHAR(20) NOT NULL,
   DOCUMENTO_AFILIADO VARCHAR(20) NOT NULL,
   FECHA_AFILIACION DATE NOT NULL,
   TELEFONO_AFILIADO VARCHAR(15)NOT NULL,
   ID_DIRECCION_FK VARCHAR (15) NOT NULL,
   ID_CLINICA_FK VARCHAR(15)NOT NULL
   )

   CREATE TABLE GRUPAL(
   ID_GRUPAL VARCHAR(15) PRIMARY KEY NOT NULL,
   NOMBRE_COMBENIO VARCHAR(25) NOT NULL,
   PORCENTAGE_REBAJA DECIMAL,
   ID_AFILIADO_FK VARCHAR(15) NOT NULL,
 )

  CREATE TABLE DIRECCION (
  ID_DIRECCION VARCHAR(15) PRIMARY KEY NOT NULL,
  DEPARTAMENTO_DIRECCION VARCHAR(25) NOT NULL,
  CIUDAD_DIRECCION VARCHAR(25) NOT NULL,
  CALLE_DIRECCION VARCHAR(15) NOT NULL,
  NUMERO_DIRECCION VARCHAR(25) NOT NULL,
  FECHA_INICIO DATE NOT NULL,
  ESTADO_TRATAMIENTO VARCHAR(25) NOT NULL,
  )

  CREATE TABLE TRATAMIENTO(
  ID_TRATAMIENTO VARCHAR(15) PRIMARY KEY NOT NULL,
  NOMBRE_TRATAMIENTO VARCHAR(20) NOT NULL,
  COSTO INTEGER NOT NULL,
  FECHA_FIN DATE NOT NULL,
  ID_CLINICA_FK VARCHAR(15) NOT NULL
)
  
  CREATE TABLE ODONTOLOGOS(
  ID_ODONTOLOGOS VARCHAR(25) PRIMARY KEY NOT NULL,
  TIPO_DOCUMENTO_ODONTOLOGO VARCHAR(25) NOT NULL,
  DOCUMENTO_ODONTOLOGO VARCHAR (15) NOT NULL,
  NOMBRE_ODONTOLOGO VARCHAR(15) NOT NULL,
  APELLIDO_ODONTOLOGO VARCHAR(25) NOT NULL,
  TELEFONO_ODONTOLOGO VARCHAR(15) NOT NULL,
  ESPECIALIDAD VARCHAR(35) NOT NULL,
  ID_DIRECCION_FK VARCHAR(15) NOT NULL,
  )

  CREATE TABLE HORARIO(
  ID_HORARIO VARCHAR(15) PRIMARY KEY NOT NULL,
  DIA VARCHAR(10) NOT NULL,
  HORA_COMIENZO VARCHAR(10) NOT NULL,
  HORA_FINAL VARCHAR(10) NOT NULL,
  ID_LOCAL_FK VARCHAR(15) NOT NULL,
  ID_ODONTOLOGO_FK VARCHAR(15) NOT NULL
  
  )

  CREATE TABLE CONSULTA(
  ID_CONSULTA VARCHAR(15) PRIMARY KEY NOT NULL,
  FECHA_CONSULTA DATE NOT NULL,
  ID_TRATAMIENTO_FK VARCHAR(15) NOT NULL,
  ID_HORARIO_FK VARCHAR(15) NOT NULL,
  DESCRIPCION_CONSULTA VARCHAR(35) NOT NULL
  )

  CREATE TABLE HISTORIA_CLINICA(
  ID_HISTORIA_CLINICA VARCHAR(15) PRIMARY KEY NOT NULL,
  ID_CONSULTA VARCHAR(15)
  )

  CREATE TABLE CUENTA_CORRIENTE(
